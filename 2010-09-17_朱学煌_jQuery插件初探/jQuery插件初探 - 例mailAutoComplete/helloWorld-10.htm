<script>
$.fn.demo = function(options) {    
	var defaults = {    
		'name' : '浙江中塑在线',    
		'class' : 'red'   
	};    
	// Extend our default options with those provided.    
	var opts = $.extend(defaults, options);    
	
	// Our plugin code write here.
	this.each(function() {
		$(this).addClass(opts.class);
		$(this).html(opts.name);
	});	
} 
$('#demo').demo({    
	'class' : 'blue'    
});
</script>
