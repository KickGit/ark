<script>
var settings = { validate:false, limit:5, name:"foo" }; 
$.each(settings, function(k, v) {
	alert('原来值：' + k + ' => ' + v);	
});
var options = { validate:true, name:"bar" };
jQuery.extend(settings, options); 
//settings == { validate: true, limit: 5, name: "bar" }
$.each(settings, function(k, v) {
	alert('更新值：' + k + ' => ' + v);	
});
</script>
